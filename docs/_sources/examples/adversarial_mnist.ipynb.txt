    "x = m.addMVar(example.shape, lb=0.0, ub=1.0, name=\"X\")\n",

    "output = m.addMVar(output_shape, lb=-gp.GRB.INFINITY, name=\"y\")"

    "\n",
    "# Bound on the distance to example in norm-1\n",
    "m.addConstr(absdiff[0, :] >= x[0, :] - example.to_numpy()[0, :])\n",
    "m.addConstr(absdiff[0, :] >= -x[0, :] + example.to_numpy()[0, :])\n",
    "m.addConstr(absdiff[0, :].sum() <= epsilon)"
